window.jumpLessonStep=function(id,step){const s=state.sessions.find(x=>String(x.id)===String(id));if(!s)return;if(step<=s.lesson.step||s.status==='completed'){s.lesson.step=step;save();render()}};
window.lessonBack=function(id){const s=state.sessions.find(x=>String(x.id)===String(id));if(!s||s.lesson.step===0)return;clearLessonTimer();s.lesson.step--;save();render()};
window.lessonContinue=function(id){
const s=state.sessions.find(x=>String(x.id)===String(id));if(!s)return;clearLessonTimer();
if(s.lesson.step===2&&!response(s,'verify').trim()){toast('Add your correction or confirmation first.');return}
if(s.lesson.step===2){s.lesson.step=3;s.lesson.handoff=true;s.lesson.currentSpeaker='partner';save();render();return}
if(s.lesson.step===3){s.lesson.currentSpeaker='self'}
if(s.lesson.step===7&&s.lesson.exerciseRound<3){s.lesson.exerciseRound++;s.lesson.timerSeconds=90;s.lesson.currentSpeaker=s.lesson.exerciseRound<2?'partner':'self';save();render();return}
if(s.lesson.step===MODULES.length-1){completeLesson(s);return}
s.lesson.step++;save();render();
};
window.completeHandoff=function(id){const s=state.sessions.find(x=>String(x.id)===String(id));if(!s)return;s.lesson.handoff=false;save();render()};
window.saveAndExit=function(id){save();navigate('/sessions');toast('Session progress saved.')};
